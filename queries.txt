#To select disease name from orpha number.
SELECT Disease.disease_name FROM Correlation INNER JOIN Disease ON Correlation.disease_orpha=Disease.orpha_number;

#To select disease name and symptoms from orpha_id:
SELECT Disease.disease_name, Symptom.symptom_name FROM Correlation INNER JOIN Disease ON Correlation.disease_orpha=Disease.orpha_number INNER JOIN Symptom ON Correlation.symptom_id = Symptom.id;
